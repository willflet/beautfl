    def __init__(self, path):

    @cached_property
    def root(self):
        return self._get_root()

    def _get_root(self):
        with open(self.path, 'r') as f:
            root = json.load(f)
        return root


    @property
    def stations(self):
        return self._get_stations()

    def _get_stations(self):
        for e in self.root:
            yield self.DisusedStation(e)
