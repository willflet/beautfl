    def _get_root(self):
        with open(self.path, 'r') as f:
            root = json.load(f)['2020']
        return root

    @property
    def stations(self):
        return self._get_stations()

    def _get_stations(self):
        stations = set()
        for osi in self.osis:
            stations.update({osi.a, osi.b})
        return stations

    @property
    def osis(self):
        return self._get_osis()

    def _get_osis(self):
        for osi in self.root:
            yield self._OSI(osi)
